import numpy as np
import matplotlib.pyplot as plt

#   ( 7)
x_train_7 = np.array([
    [23, 31],
    [25, 22],
    [32, 11],
    [8, 34],
    [24, 31],
    [9, 37],
    [23, 35],
    [23, 42],
    [38, 19]
])

y_train_7 = np.array([1, 1, -1, 1, -1, -1, 1, -1, 1])

#  
x_train = np.array([x.tolist() + [10 * x[0], 10 * x[1], 5 * (x[0] + x[1])] for x in x_train_7])
y_train = y_train_7

def loss(w, x, y):
    """Sigmoid loss function."""
    M = np.dot(w, x) * y
    return 2 / (1 + np.exp(M))

def df(w, x, y):
    """Derivative of sigmoid loss function with L2 regularization."""
    L2 = 1.0
    M = np.dot(w, x) * y
    grad_loss = -2 * (1 + np.exp(M))**(-2) * np.exp(M) * x * y
    grad_reg = 2 * L2 * w
    return grad_loss + grad_reg

# Training setup
n_features = len(x_train[0])
n_train = len(x_train)
w = np.zeros(n_features)

nt = 0.00001
lm = 0.01
N = 5000

Q = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
Q_plot = [Q]

# Optimization loop
for i in range(N):
    k = np.random.randint(0, n_train)
    ek = loss(w, x_train[k], y_train[k])
    w = w - nt * df(w, x_train[k], y_train[k])
    Q = lm * ek + (1 - lm) * Q
    Q_plot.append(Q)

Q_final = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
final_weights = w

# Plot training loss
plt.figure(figsize=(10, 5))
plt.plot(Q_plot, label="Empirical Risk (L2)")
plt.xlabel("Iteration")
plt.ylabel("Risk")
plt.title("Risk Evolution Over Iterations (L2 Regularization, Variant 7)")
plt.legend()
plt.grid(True)
plt.show()

# Predictions
predictions = np.array([np.sign(np.dot(w, x)) for x in x_train])

# Visualization
plt.figure(figsize=(10, 6))
for point, true_label, pred_label in zip(x_train_7, y_train, predictions):

import numpy as np

x_train = np.array([[0.5, 1.0],
                    [1.5, 2.0],
                    [3.0, 4.0],
                    [2.5, 3.0]])
y_train = np.array([1, 1, -1, -1])

X = np.hstack([np.ones((x_train.shape[0], 1)), x_train])

w = np.zeros(X.shape[1])

learning_rate = 0.01
n_iterations = 1000

for i in range(n_iterations):
    idx = np.random.randint(0, X.shape[0])
    xi = X[idx]
    yi = y_train[idx]
    
    if yi * np.dot(w, xi) <= 0:
        w = w + learning_rate * yi * xi

print(" :", w)

def predict(X, w):
    return np.where(np.dot(X, w) >= 0, 1, -1)

predictions = predict(X, w)
print(" :", predictions)
